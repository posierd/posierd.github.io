import requests
import pprint
import csv


for page in range(1,6):
    print("========正在获取{}页数据========1".format(page))
    url = 'http://www.kfc.com.cn/kfccda/ashx/GetStoreList.ashx?op=keyword'
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.163 Safari/537.36'
    }
    data = {
        'cname':'',
        'pid':'',
        'keyword':'北京',
        'pageIndex':str(page),
        'pageSize':'10',
    }
    request = requests.post(url,headers=headers,data=data)
    html = request.json()
    # pprint.pprint(html)

    json_data = html['Table1']
    # pprint.pprint(json_data)
    for a in json_data:
        name = a ['storeName']
        city = a['cityName']
        ip = a['addressDetail']
        pro = a['pro']
        # print(name,city,ip,pro)


        path = r'C:\Users\DELL\Desktop\python_wd\肯德基.csv'
        print('爬取：',name)
        with open(path,mode='a',newline='')as csvfile:
            csvwrite = csv.writer(csvfile,delimiter=',' )
            csvwrite.writerow([name,city,ip,pro])
