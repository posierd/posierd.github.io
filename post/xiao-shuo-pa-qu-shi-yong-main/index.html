#  目标地址： https://wx.tianyabooks.com/book/hywx5/
import re
import requests
def get_url(url):
   r = requests.get(url)
   r.encoding = r.apparent_encoding
   return r.text
def get_messages(html):
   all_message = re.findall(r'<dd><a href="(.*?)">(.*?)</a></dd>',html)
   return all_message




def get_content(url):
   r = get_url(url)
   con = re.findall(r'<td width="760" align="left">(.*?)<td width="20"></td>',r, re.S)[0]
   con = con.replace('&nbsp;&nbsp;&nbsp;&nbsp;', '')
   con = con.replace('<br />', '')
   con = con.replace('</p>', '')
   con = con.replace('</td>', '')
   con = con.replace('<p>','')
   return con



def main():
   url = 'https://wx.tianyabooks.com/book/hywx5/'
   html = get_url(url)  #  获取主页面的章节信息（这是一个函数）
   # print(html)
   # exit()
   content = get_messages(html)  # 获取所有章节的短 url  和 章节名字
   # print(content)
   # exit()
   title = re.findall(r'<h1>《(.*?)》</h1>',html)[0].strip()
#    print(title)  找出小说的名字





   with open(r'C:\Users\DELL\Desktop\python_wd\文本信息\{}.txt'.format(title),'a+',encoding='utf_8')as f:
       for mulu in content:
           f.write("\n\t\t")
           f.write(mulu[1])  #  写入章节目录
           f.write("\n")
           contents = get_content(url+mulu[0])  # 在建立一个函数 用于小说内容的获取与清洗
           f.write("\n\t")
           f.write(contents)    #  写入章节内容
           f.write("\n")

if __name__ == '__main__':
   main()
