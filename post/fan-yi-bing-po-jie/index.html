import requests
import json
#  这是一个 微软的 翻译
#  通过  newwork 找到的 url 不用改变
def  fanyi():
    link = 'https://cn.bing.com/ttranslatev3?isVertical=1&&IG=1AEDDB7B1F384527B91AE6D95104A381&IID=translator.5028.3'
    content = input("请输入需要翻译的内容: ")
    headers = {
        'user-agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.132 Safari/537.36'
    }
    data = {
        'fromLang': 'auto-detect',
        'text': content,
        'to': 'en',
    }
    resp = requests.post(link,data=data,headers=headers).json()
    print(resp[0]['translations'][0]['text'])
'''
发现这个只能翻译中文
'''
fanyi()
