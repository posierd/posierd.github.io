六房短视频获取

2020-03-28

😃😃😃

import requests
import json
import re

headers = {
    'user-agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.132 Safari/537.36'
}
def chenk_title(title):
    '''替换非法字符窜'''
    pattern = re.compile(r"[\/\\\:\*\?\"\<\>\|]")
    new_title = re.sub(pattern,"_",title)
    return new_title



for page in range(1,5):
    print("========正在获取{}页数据=========".format(page))
    link = 'https://v.6.cn/minivideo/getlist.php?act=recommend&page={}&pagesize=20'.format(page)


    resp_html = requests.get(link,headers=headers).text

    html = json.loads(resp_html)   #  数据转换  字典

    src_list = html['content']['list']

    for src in src_list:
        mp4_name = src['title']+ '.mp4'
        mp4_link = src['playurl']



        new_name = chenk_title(mp4_name)
        print("正在下载",mp4_name)
        mp4 = requests.get(mp4_link,headers=headers).content

        with open(r'C:\Users\DELL\Desktop\python_wd\shipin\六间房\\'+ new_name ,'wb')as f:
            f.write(mp4)
            print("下载完成。。。。。。\n")
