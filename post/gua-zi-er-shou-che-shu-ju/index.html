    html = etree.HTML(text)
    ul = html.xpath('//ul[@class="carlist clearfix js-top"]')[0]
    lis = ul.xpath('./li')
    detail_url_list = []
    for li in lis:
        detail_url = li.xpath('./a/@href')
        detail_url = 'https://www.guazi.com' + detail_url[0]
        detail_url_list.append(detail_url)
    return detail_url_list

#  提取数据
def parse_html(url):
    resp_2 = requests.get(url,headers=headers)
    text = resp_2.content.decode('utf-8')
    html = etree.HTML(text)
    #  车名
    title = html.xpath('//div[@class="product-textbox"]/h2/text()')[0]
    title = title.strip()    # 默认删除收尾空白
    # 信息
    info = html.xpath('//ul[@class="assort clearfix"]/li/span/text()')
    #  价格
    pic = html.xpath('//div[@class="price-main"]/span/text()')[0]


    # 根据索引取值
    infos = {}
    cardtime = info[0]
    km = info[1]
    displacement = info[2]
    speedbox = info[3]


    # 存入字典 定义键名  value  为数据
    infos['title'] = title
    infos['cardtime'] = cardtime
    infos['km'] = km
    infos['displacement'] = displacement
    infos['speedbox'] = speedbox
    infos['pic'] = pic
    print(infos)
    return infos

#  保存数据
def save_file(infos,f):
    f.write('{},{},{},{},{},{}\n'.format(infos['title'],infos['cardtime'],infos['km'],infos['displacement'],infos['speedbox'],infos['pic']))

#   主函数  程序入口
