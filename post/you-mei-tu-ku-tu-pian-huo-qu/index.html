优美图库图片获取

2020-04-01

😃😃😃

import requests
import parsel

headers = {
    'user-agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.149 Safari/537.36'
}

name = 1
def down_img(img_url):
    global name
    for new_url in img_url:
        resp_3 = requests.get(new_url,headers=headers)
        print("\t正在下载第{}张图片".format(name))
        path = r'C:\Users\DELL\Desktop\python_wd\mig\优美图片\{}.jpg'.format(name)
        with open(path,'wb')as f:
            f.write(resp_3.content)
            name += 1

def main(page):
    print("正在下载第{}页数据".format(page))
    url = 'https://www.umei.cc/p/gaoqing/cn/{}.htm'.format(page)
    resp = requests.get(url,headers=headers)
    resp.encoding = resp.apparent_encoding
    data = parsel.Selector(resp.text)
    url_1 = data.xpath('//div[@class="TypeList"]/ul/li/a/@href').extract()
    for a in url_1:
        resp_2 = requests.get(a,headers=headers)
        resp_2.encoding = resp_2.apparent_encoding
        data_2 = parsel.Selector(resp_2.text)
        img_url = data_2.xpath('//div[@class="ImageBody"]/p/a/img/@src').extract()
        down_img(img_url)

if __name__ == '__main__':
    num = int(input("请输入需要获取的页数："))
    for page in range(1,num +1):
        main(page)
