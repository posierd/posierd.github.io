博客标题爬取

2020-03-06

😃😃
目标地址：http://www.santostang.com/

import requests
from bs4 import BeautifulSoup
link = 'http://www.santostang.com/'
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/79.0.3945.130 Safari/537.36'
}
r = requests.get(link,headers=headers)
print(r.status_code)
soup = BeautifulSoup(r.text,'html.parser')
title_list = soup.find_all('article',class_='article-list-1 clearfix')
#print(title_list)
for i in title_list:
    title = i.find('h1',class_='post-title').get_text()
    print(title)
    with open(r'C:\Users\DELL\Desktop\python_wd\文本信息\唐松博客标题.txt','a+')as f:
        f.write('\n')
        f.write(title)
        f.write('\n')
        f.close()
