import requests
import re
from urllib.request import urlretrieve
'''
这是一部分的思路
获取页面源代码  获取视频id---- video_1661731
拼接完整的 url 链接---- https://www.pearvideo.com/video_1661731
发送请求  在获取其中的源代码  找出视频的真实播放地址
再次发送 视频真实地址 --https://video.pearvideo.com/mp4/third/20200316/cont-1661859-11107897-203035-hd.mp4
二进制 保存本地

'''

headers = {
        'user-agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.132 Safari/537.36'
    }
def down_mp4():
    # 获取 id
    link = 'https://www.pearvideo.com/category_6'
    resp = requests.get(link,headers=headers)
    resp.encoding = resp.apparent_encoding
    html = resp.text
    red = r'<a href="(.*?)" class="vervideo-lilink actplay">'
    vidle_id = re.findall(red,html)
    print("获取到的视频数量:",len(vidle_id))

    #  拼接id  存入列表
    link_list = []  #  存入拼接好的 link
    start_link = 'https://www.pearvideo.com/'
    for vidle in  vidle_id:
        new_link = start_link + vidle
        link_list.append(new_link)

    # 获取视频的真实播放地址
    for link_two in link_list:
        resp_two = requests.get(link_two,headers=headers).text
        # 播放地址
        rea = r'sdUrl="",ldUrl="",srcUrl="(.*?)",vdoUrl'
        play_val = re.findall(rea,resp_two)
        # 视频标题
        reb = r'<h1 class="video-tt">(.*?)</h1>'
        vidle_name = re.findall(reb,resp_two)
        # 进行视频保存
        print("\n正在下载视频:{}".format(vidle_name[0]))
        file_path =r'C:\Users\DELL\Desktop\python_wd\shipin\梨视频\\'
        path = file_path + '%s.mp4'%vidle_name[0]
        urlretrieve(play_val[0],path)
        print("\n视频下载完成。。。。。。")


down_mp4()
