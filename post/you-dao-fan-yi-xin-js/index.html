import requests
import time
import random   #  随机数
import hashlib  #  md5 加密



headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.163 Safari/537.36',
    'Referer': 'http://fanyi.youdao.com/',
    'Cookie': 'OUTFOX_SEARCH_USER_ID=-148473857@10.169.0.83; JSESSIONID=aaafbpaaZm1Q5-_wxwLgx; OUTFOX_SEARCH_USER_ID_NCOO=982336911.512214; ___rl__test__cookies=1587617780880',
}



#  输入翻译内容
word = input("请输入需要翻译的内容：")


#   请求时携带的参数
data = {
    'i':word,
    'from': 'AUTO',
    'to': 'AUTO',
    'smartresult': 'dict',
    'client': 'fanyideskweb',
    #  时间戳
    'salt':None,
    'sign': None,
    # 时间戳（毫秒）
    'ts': None,
    'bv': None,
    'doctype': 'json',
    'version': '2.1',
    'keyfrom': 'fanyi.web',
    'action': 'FY_BY_CLICKBUTTION',
}


'''
关于时间戳 查找在线工具（删除后4位 s单位）来转换
后4位为随机数
参数  sign  bv  是变化的（是本地js加密）
在网页源代码中找到  </script>  段落 复制内容进行格式化将内容本地 保存为 xxx.js 文件
在保存的文件中 用 data  的 参数client 里的值  查找  
分析 代码
sign: n.md5("fanyideskweb" + e + i + "Nw(nmmbP%A-r6U3EUn]Aj")  
 e  为翻译的内容
 i  为 参数 salt 的值 ——   r + 时间戳
 .....
 就用这段以 python 来模拟
 
'''

#   根据查找数据 找到这段
#   将当中的内容用 python  来实现
'''
var t = n.md5(navigator.appVersion),
r = "" + (new Date).getTime(),
i = r + parseInt(10 * Math.random(), 10);
return {   
    ts: r,
    bv: t,
    salt: i,
    sign: n.md5("fanyideskweb" + e + i + "Nw(nmmbP%A-r6U3EUn]Aj")
}
'''
#   开始实现

#  时间戳加密
ts = str(int(time.time()) * 1000)
salt = ts + str(int(random.random()*10))

#  拼接
sign = "fanyideskweb" + word + salt + "Nw(nmmbP%A-r6U3EUn]Aj"
#  处理（加密）
sign = hashlib.md5(sign.encode('utf-8')).hexdigest()


#  请求头
bv = "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.163 Safari/537.36"
#  处理（加密）
bv = hashlib.md5(bv.encode('utf-8')).hexdigest()

# 修改携带参数  通过键名来修改值（上面模拟的参数）
data['i'] = word
data['ts'] = ts
data['salt'] = salt
data['sign'] = sign
data['bv'] = bv



#  发送请求  提取数据
url = 'http://fanyi.youdao.com/translate_o?smartresult=dict&smartresult=rule'
resp_1 = requests.post(url,data=data,headers=headers)
html = resp_1.json()['translateResult']
for a in html:
    mess = a[0]['tgt']
    print("输出结果：",mess)
